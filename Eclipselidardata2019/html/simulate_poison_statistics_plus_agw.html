mean_OH_counts = 1300;
phase = 0 * pi / 180;

lx = 512; % km
ly = 1e6; % km

kx = 2*pi/lx;
ky = 2*pi/ly;

I_prime = 0.1;


[x,y] = meshgrid(1:512, 1:512);

w = I_prime .* sin(kx.*x + ky.*y + phase);

I_hat = poissrnd(mean_OH_counts,[512 512]);

I = I_hat .* (1+w);

imagesc(I)
axis square
colormap gray
